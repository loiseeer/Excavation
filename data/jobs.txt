mission `Outpost supply run`
	name `Outpost supply run`
	description `Deliver <cargo> to <destination>, The payment is <payment>` 
	blocked 'You need <capacity> to do this job'
	job
	repeat
	to offer
		random > 10
	source "Newmaker"
	destination "Helix Outpost"
	cargo `supply` 60 110
	on complete
		payment 0 700
		dialog "You unloaded <cargo> and a payment of <payment> was transfered to your account"

mission `Outpost supply run (Escort)`
	name `Supply ship escort`
	description `A ship captain carrying supply to <destination> and willing to pay <payment> for additional protection`
	job
	repeat
	to offer
		random > 20
		"combat rating" > 4
	source "Kyuki"
	destination "Helix Outpost"
	npc save accompany
		personality
			timid opportunistic
			confusion 30
		fleet "Helix Maintenance" 1
	on complete
		payment 150000
		dialog `The ship captain thank you for your service and paid you <payment>`
	on visit
		dialog phrase "generic arrived-without-npc dialog"

mission `Relic recon (1)`
	name `Relic recon`
	description `Scout <waypoints> and report the data back at <destination>, The payment is <payment>`
	job
	repeat
	to offer
		random > 10
	waypoint
		not government "Helix Corp"
		neighbor distance 3 5
	waypoint
		not government "Helix Corp"
		neighbor distance 2 4
	source
		government "Helix Corp"
	destination
	on complete
		payment 400000
		dialog `You transfered the data to the station and received <payment>`

mission `Relic threat`
	name `Supress Relic fleets`
	description `A small relic fleet have been spotted nearby, neutralize the fleet before it can cause any damage and then return to <planet> for a payment of <payment>`
	job
	repeat
	to offer
		random >= 80
		"combat rating" > 8
	npc disable
		government "Relic"
		personality
			opportunistic heroic staying target
			confusion 30
		system
			not government "Helix Corp"
			neighbor distance 1 2
		fleet "Relic Light" 2
		dialog "You have successfully neutralized the Relic fleet, time to return to <destination> for the payment"
	source "Kyuki"
	destination "Kyuki"
	on complete
		payment 250000
		dialog "A Helix staff thank you for your work and transfered <payment> to your account"

mission `Weapon research (Caster)`
	name `Weapon research (Bolt Caster)`
	description `Helix researcher is looking for a pair of Bolt Caster and are willing to pay <payment> for it`
